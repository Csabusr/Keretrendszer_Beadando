package hu.uni.eku.tzs.controller;

import hu.uni.eku.tzs.controller.dto.MovieDirectorDto;
import hu.uni.eku.tzs.controller.dto.MovieDirectorMapper;
import hu.uni.eku.tzs.model.MovieDirector;
import hu.uni.eku.tzs.service.MovieDirectorManager;
import hu.uni.eku.tzs.service.exceptions.MovieDirectorAlreadyExistsException;
import hu.uni.eku.tzs.service.exceptions.MovieDirectorNotFoundException;
import io.swagger.annotations.Api;
import io.swagger.annotations.ApiOperation;
import lombok.RequiredArgsConstructor;
import org.springframework.http.HttpStatus;
import org.springframework.web.bind.annotation.DeleteMapping;
import org.springframework.web.bind.annotation.GetMapping;
import org.springframework.web.bind.annotation.PathVariable;
import org.springframework.web.bind.annotation.PostMapping;
import org.springframework.web.bind.annotation.PutMapping;
import org.springframework.web.bind.annotation.RequestBody;
import org.springframework.web.bind.annotation.RequestMapping;
import org.springframework.web.bind.annotation.RequestParam;
import org.springframework.web.bind.annotation.RestController;
import org.springframework.web.server.ResponseStatusException;

import javax.validation.Valid;
import java.util.Collection;
import java.util.stream.Collectors;

@Api(tags = "Movies_Directors")
@RequestMapping("/movies_directors")
@RestController
@RequiredArgsConstructor
public class MovieDirectorController {

    private final MovieDirectorManager movieDirectorManager;

    private final MovieDirectorMapper movieDirectorMapper;

    @ApiOperation("Read All")
    @GetMapping(value = {"/", ""})
    public Collection<MovieDirectorDto> readAllDirectors() {
        return movieDirectorManager.readAll()
                .stream()
                .map(movieDirectorMapper::movieDirector2MovieDirectorDto)
                .collect(Collectors.toList());
    }

    @ApiOperation("Record")
    @PostMapping(value = {"", "/"})
    public MovieDirectorDto create(@Valid @RequestBody MovieDirectorDto recordRequestDto) {
        MovieDirector movieDirector = movieDirectorMapper.movieDirectorDto2MovieDirector(recordRequestDto);
        try {
            MovieDirector recordedMovieDirector = movieDirectorManager.record(movieDirector);
            return movieDirectorMapper.movieDirector2MovieDirectorDto(recordedMovieDirector);
        } catch (MovieDirectorAlreadyExistsException e) {
            throw new ResponseStatusException(HttpStatus.INTERNAL_SERVER_ERROR, e.getMessage());
        }
    }

    @ApiOperation("Update")
    @PutMapping(value = {"", "/"})
    public MovieDirectorDto update(@Valid @RequestBody MovieDirectorDto updateRequestDto) {
        MovieDirector movieDirector = movieDirectorMapper.movieDirectorDto2MovieDirector(updateRequestDto);
        MovieDirector updatedMovieDirector = movieDirectorManager.modify(movieDirector);
        return movieDirectorMapper.movieDirector2MovieDirectorDto(updatedMovieDirector);
    }

    @ApiOperation("Delete")
    @DeleteMapping(value = {"", "/"})
    public void delete(@RequestParam int id) {
        try {
            movieDirectorManager.delete(movieDirectorManager.readById(id));
        } catch (MovieDirectorNotFoundException e) {
            throw new ResponseStatusException(HttpStatus.INTERNAL_SERVER_ERROR, e.getMessage());
        }
    }

    @ApiOperation("Delete")
    @DeleteMapping(value = {"/{id}"})
    public void deleteBasedOnPath(@PathVariable int id) {
        this.delete(id);
    }
}
