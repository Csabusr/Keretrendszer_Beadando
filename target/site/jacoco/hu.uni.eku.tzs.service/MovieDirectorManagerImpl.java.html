package hu.uni.eku.tzs.service;

import hu.uni.eku.tzs.dao.MovieDirectorRepository;
import hu.uni.eku.tzs.dao.entity.DirectorEntity;
import hu.uni.eku.tzs.dao.entity.MovieDirectorEntity;
import hu.uni.eku.tzs.dao.entity.MovieEntity;
import hu.uni.eku.tzs.model.Director;
import hu.uni.eku.tzs.model.Movie;
import hu.uni.eku.tzs.model.MovieDirector;
import hu.uni.eku.tzs.service.exceptions.MovieDirectorAlreadyExistsException;
import hu.uni.eku.tzs.service.exceptions.MovieDirectorNotFoundException;
import lombok.RequiredArgsConstructor;
import org.springframework.data.domain.PageRequest;
import org.springframework.stereotype.Service;

import java.util.Collection;
import java.util.Optional;
import java.util.stream.Collectors;

@Service
@RequiredArgsConstructor
public class MovieDirectorManagerImpl implements MovieDirectorManager {

    private final MovieDirectorRepository movieDirectorRepository;

    @Override
    public Collection<MovieDirector> readAll() {
        return movieDirectorRepository.findAll(PageRequest.of(0, 100)).stream()
                .map(MovieDirectorManagerImpl::convertMovieDirectorEntity2Model)
                .collect(Collectors.toList());
    }

    @Override
    public MovieDirector readById(int id) throws MovieDirectorNotFoundException {
        Optional<MovieDirectorEntity> entity = movieDirectorRepository.findById(id);
        if (entity.isEmpty()) {
            throw new MovieDirectorNotFoundException(String.format("Cannot find MovieDirector with this id %s", id));
        }

        return convertMovieDirectorEntity2Model(entity.get());
    }

    @Override
    public MovieDirector modify(MovieDirector movieDirector) {
        MovieDirectorEntity entity = convertMovieDirectorModel2Entity(movieDirector);
        return convertMovieDirectorEntity2Model(movieDirectorRepository.save(entity));
    }

    @Override
    public void delete(MovieDirector movieDirector) throws MovieDirectorNotFoundException {
        movieDirectorRepository.delete(convertMovieDirectorModel2Entity(movieDirector));
    }

    @Override
    public MovieDirector record(MovieDirector movieDirector) throws MovieDirectorAlreadyExistsException {
        if (movieDirectorRepository.findById(movieDirector.getMovie().getId()).isPresent()) {
            throw new MovieDirectorAlreadyExistsException();
        }
        MovieDirectorEntity movieDirectorEntity = movieDirectorRepository.save(
                MovieDirectorEntity.builder()
                        .movie(convertMovieModel2Entity(movieDirector.getMovie()))
                        .director(convertDirectorModel2Entity(movieDirector.getDirector()))
                        .build()
        );
        return convertMovieDirectorEntity2Model(movieDirectorEntity);
    }

    private static MovieDirector convertMovieDirectorEntity2Model(MovieDirectorEntity movieDirectorEntity) {
        return new MovieDirector(
                convertMovieEntity2Model(movieDirectorEntity.getMovie()),
                convertDirectorEntity2Model(movieDirectorEntity.getDirector())
        );
    }

    private static MovieDirectorEntity convertMovieDirectorModel2Entity(MovieDirector movieDirector) {
        return MovieDirectorEntity.builder()
                .movie(convertMovieModel2Entity(movieDirector.getMovie()))
                .director(convertDirectorModel2Entity(movieDirector.getDirector()))
                .build();
    }

    private static Director convertDirectorEntity2Model(DirectorEntity directorEntity) {
        return new Director(
                directorEntity.getId(),
                directorEntity.getFirstname(),
                directorEntity.getLastname()
        );
    }

    private static DirectorEntity convertDirectorModel2Entity(Director director) {
        return DirectorEntity.builder()
                .id(director.getId())
                .firstname(director.getFirstname())
                .lastname(director.getLastname())
                .build();
    }

    private static Movie convertMovieEntity2Model(MovieEntity movieEntity) {
        return new Movie(
                movieEntity.getId(),
                movieEntity.getName(),
                movieEntity.getYear(),
                movieEntity.getRank()
        );
    }

    private static MovieEntity convertMovieModel2Entity(Movie movie) {
        return MovieEntity.builder()
                .id(movie.getId())
                .name(movie.getName())
                .year(movie.getYear())
                .rank(movie.getRank())
                .build();
    }
}
